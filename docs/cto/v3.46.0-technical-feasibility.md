**Agent 执行引擎**:

```python
# backend/services/agent/executor.py

class AgentExecutor:
    def __init__(self, tools: List[BaseTool], llm_service):
        self.tools = {t.name: t for t in tools}
        self.llm = llm_service

    async def run(self, task: str, max_iterations: int = 10):
        """执行 Agent 任务"""
        context = {"task": task, "history": []}

        for i in range(max_iterations):
            # 1. LLM 决策下一步
            decision = await self.llm.decide_action(context)

            if decision.action == "finish":
                return decision.result

            # 2. 执行工具
            if decision.tool in self.tools:
                result = await self.tools[decision.tool].execute(**decision.args)
                context["history"].append({
                    "action": decision.action,
                    "tool": decision.tool,
                    "args": decision.args,
                    "result": result.model_dump()
                })

            # 3. 检查是否需要用户确认
            if decision.requires_confirmation:
                yield {"type": "confirmation_required", "action": decision}

        return {"error": "Max iterations reached"}
```

```python
# backend/services/mcp/client.py

from mcp import ClientSession, StdioServerParameters
from mcp.client.stdio import stdio_client

class MCPClientManager:
    """管理多个 MCP Server 连接"""

    def __init__(self):
        self.sessions: Dict[str, ClientSession] = {}
        self.tools: Dict[str, MCPTool] = {}

    async def connect_server(self, name: str, command: str, args: List[str]):
        """连接 MCP Server"""
        server_params = StdioServerParameters(
            command=command,
            args=args,
        )

        async with stdio_client(server_params) as (read, write):
            async with ClientSession(read, write) as session:
                await session.initialize()
                self.sessions[name] = session

                # 注册工具
                tools = await session.list_tools()
                for tool in tools.tools:
                    self.tools[f"{name}.{tool.name}"] = MCPTool(
                        server_name=name,
                        tool=tool,
                        session=session
                    )

    async def call_tool(self, tool_id: str, arguments: dict):
        """调用 MCP 工具"""
        if tool_id not in self.tools:
            raise ValueError(f"Tool not found: {tool_id}")

        mcp_tool = self.tools[tool_id]
        return await mcp_tool.session.call_tool(
            mcp_tool.tool.name,
            arguments
        )
```

### 3.2 智能模型路由

**技术方案**:

```python
# backend/services/model_router.py

class ModelRouter:
    """根据任务特征自动选择最优模型"""

    ROUTING_RULES = {
        "code": {
            "keywords": ["代码", "code", "debug", "function", "编程"],
            "preferred": "deepseek-chat",  # DeepSeek 代码能力强
            "fallback": "gpt-4o"
        },
        "reasoning": {
            "keywords": ["分析", "推理", "逻辑", "why", "explain"],
            "preferred": "deepseek-reasoner",  # R1 推理强
            "fallback": "claude-3-5-sonnet"
        },
        "creative": {
            "keywords": ["写作", "creative", "故事", "文章"],
            "preferred": "claude-3-5-sonnet",
            "fallback": "gpt-4o"
        },
        "simple": {
            "keywords": [],  # 默认
            "preferred": "deepseek-chat",  # 成本最优
            "fallback": "gpt-4o-mini"
        }
    }

    def classify_task(self, message: str) -> str:
        """分类任务类型"""
        message_lower = message.lower()
        for task_type, rules in self.ROUTING_RULES.items():
            if any(kw in message_lower for kw in rules["keywords"]):
                return task_type
        return "simple"

    def select_model(self, message: str, available_models: List[str]) -> str:
        """选择最优模型"""
        task_type = self.classify_task(message)
        rules = self.ROUTING_RULES[task_type]

        if rules["preferred"] in available_models:
            return rules["preferred"]
        return rules["fallback"]
```
